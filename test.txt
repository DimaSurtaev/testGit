git init - создать репозиторий
git add - подготовить изменения 
git commit - сохранить изменения и добавить в репозиторий
git checkout - перейти к определенной версии
git status - посмотерть статус файла
git log - просмотр истории изменений(коммитов)
git checkout -b - создание новой ветки и переход на нее
git branch - посмотреть все ветки
git branch -m - переименовать текущую ветку
git bracnh -d - удалить ветку (нужно выйти из ветки которую нужно удалить)
